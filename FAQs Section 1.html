$(document).ready(function() { const listName = "FAQs"; let accordion = $("#faqAccordion"); // Initialize the SharePoint library $pnp.setup({ sp: { baseUrl: "https://your-domain.sharepoint.com/sites/your-site" } }); // Fetch data from the SharePoint list and populate the Accordion $pnp.sp.web.lists.getByTitle(listName).items.select("Title", "Question", "Answer").get().then(items => { items.forEach((item, index) => { accordion.append(`

${item.Answer}
`); }); }).catch(error => { console.error("Error fetching data from the SharePoint list:", error); }); }); SharePoint FAQ Accordion
